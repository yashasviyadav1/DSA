# 662. Maximum Width of Binary Tree

### [Leetcode](https://leetcode.com/problems/maximum-width-of-binary-tree/description/)

![image](https://github.com/user-attachments/assets/bbf00ca5-83d6-44b7-819c-fbd11ffaacd2)


```java
// idea : we need the largest number of nodes present at any level in the whole tree (count the null nodes as well) so for that use indexing in tree give pos to nodes 

class Pair{
    TreeNode node;
    int position; // its position in the tree going left to right from top to bottom
    Pair(TreeNode node, int position){
        this.node = node;
        this.position = position;
    }
}
class Solution {
    public int widthOfBinaryTree(TreeNode root) {
        
        // dq of (node, its position) dq because at any level we need to extract its first and last node for width calc
        Deque<Pair> dq = new ArrayDeque<>();
        dq.addLast(new Pair(root, 0));

        int maxWidth = 0;
        while(!dq.isEmpty()){
            
            int width = dq.getLast().position - dq.getFirst().position + 1; // calculating width of current level
            maxWidth = Math.max(maxWidth, width);

            int size = dq.size();
            for(int i=0; i < size; i++){
                Pair p = dq.removeFirst();
                TreeNode frontNode = p.node;
                int pos = p.position;
                // for ith node left child is at at 2*i + 1 index & right child at 2*i + 2 index
                if(frontNode.left != null) dq.addLast(new Pair(frontNode.left, 2*pos + 1)); // l
                if(frontNode.right != null) dq.addLast(new Pair(frontNode.right, 2*pos + 2));
            }
        }

        return maxWidth;
    }
}
```
